{{- /* Renders .Content with root-absolute internal links (href="/..." / src="/...") rewritten to respect baseURL's path prefix. Content bodies are raw HTML with hardcoded site-root paths (e.g. href="/projects/cover/"), which Hugo does not rewrite - this breaks when the site is served under a subpath such as https://nataliatixo.github.io/nataliatixo-web/. When baseURL sits at a domain root the prefix is "/" and this is a no-op. */ -}} {{- /* NB: relURL treats leading-slash input as already server-root-relative and returns it unchanged, so the prefix must come from baseURL itself. */ -}} {{- $prefix := (urls.Parse site.BaseURL).Path | default "/" -}} {{- if not (strings.HasSuffix $prefix "/") }}{{ $prefix = printf "%s/" $prefix }}{{ end -}} {{- $c := .Content | string -}} {{- if ne $prefix "/" -}} {{- /* Shortcode output ({{< img >}}, RelPermalinks) is already prefixed; shield it from the rewrite (RE2 has no lookahead), prefix the rest, then restore. The [^/] guard skips protocol-relative //host URLs. */ -}} {{- $c = replace $c (printf `="%s` $prefix) `="__PREFIXED__/` -}} {{- $c = replaceRE `(href|src)="/([^/])` (printf `${1}="%s${2}` $prefix) $c -}} {{- /* Bare href="/" (home link) — the [^/] guard above skips it. */ -}} {{- $c = replaceRE `(href|src)="/"` (printf `${1}="%s"` $prefix) $c -}} {{- $c = replace $c `="__PREFIXED__/` (printf `="%s` $prefix) -}} {{- end -}} {{- $c | safeHTML -}}
