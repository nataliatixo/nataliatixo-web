{{- /* Renders .Content with root-absolute internal links (href="/..." / src="/...") rewritten to respect baseURL's path prefix. Content bodies are raw HTML with hardcoded site-root paths (e.g. href="/projects/cover/"), which Hugo does not rewrite - this breaks when the site is served under a subpath such as https://nataliatixo.github.io/nataliatixo-web/. When baseURL sits at a domain root the prefix is "/" and this is a no-op. */ -}} {{- $prefix := "/" | relURL -}} {{- .Content | string | replaceRE `(href|src)="/([^/])` (printf `${1}="%s${2}` $prefix) | safeHTML -}}
