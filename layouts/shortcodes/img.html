{{- /* Body image: {{< img src="file.jpg" alt="..." >}} where src is a page-bundle resource. Serves a WebP capped at 1400px wide (with a 700px srcset variant for small screens), plus width/height and lazy loading. GIFs (animated) and srcs that don't resolve to a bundle resource pass through untouched. alt values are stored entity-escaped by extract_content.py, hence htmlUnescape. */ -}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" | default "" | htmlUnescape -}} {{- $img := .Page.Resources.Get $src -}} {{- if and $img (ne $img.MediaType.SubType "gif") -}} {{- /* Apply EXIF rotation first: Hugo strips EXIF without applying it, so camera uploads would otherwise render sideways. */ -}} {{- $img = $img.Filter images.AutoOrient -}} {{- $full := $img -}} {{- if gt $img.Width 1400 -}} {{- $full = $img.Resize "1400x webp q82" -}} {{- else -}} {{- $full = $img.Process "webp q82" -}} {{- end -}} {{- $srcset := "" -}} {{- if gt $full.Width 700 -}} {{- $small := $img.Resize "700x webp q82" -}} {{- $srcset = printf "%s 700w, %s %dw" $small.RelPermalink $full.RelPermalink $full.Width -}} {{- end -}} {{- else if $img -}} {{- else -}} {{- end -}}
